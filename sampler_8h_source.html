481 template <class Space, class Params>
482 void sampler<Space,Params>::Initialise(void)
483 {
484 T = 0;
485 dlogNCPath = 0.0;
486 acceptProb = -1;
487
488 //Set the initial values and log weights of the particles
489 std::vector<Space> InitVal(N);
490 arma::vec InitWeights(N);
491 pPopulation = population<Space>(InitVal,InitWeights);
492 pMoves->DoInit(pPopulation,N,algParams);
493
494 //Scaling weights by 1/N (for evidence computation)
495 pPopulation.SetLogWeight(pPopulation.GetLogWeight() - log(static_cast<double>(N)));
496
497 //Estimate the normalising constant
498 dlogNCIt = CalcLogNC();
499 dlogNCPath += dlogNCIt;
500
501 //Normalise the weights
502 pPopulation.SetLogWeight(pPopulation.GetLogWeight() - dlogNCIt);
503
504 //Check if the ESS is below some reasonable threshold and resample if necessary.
505 //A mechanism for setting this threshold is required.
506 double ESS = GetESS();
507 if(ESS < dResampleThreshold) {
508 nResampled = 1;
509 pAdapt->updateForMCMC(algParams,pPopulation,acceptProb,nResampled,nRepeats);
510 Resample(rtResampleMode);
511 }
512 else {
513 nResampled = 0;
514 if(htHistoryMode == HistoryType::AL) {
515 uRSIndices = arma::linspace<arma::Col<unsigned int>>(0, N - 1, N);
516 }
517 pAdapt->updateForMCMC(algParams,pPopulation,acceptProb,nResampled,nRepeats);
518 }
519
520 //A possible MCMC step should be included here.
521 bool didMCMC = pMoves->DoMCMC(0,pPopulation, N, nRepeats, nAccepted,algParams);
522 if (didMCMC){
523 acceptProb = static_cast<double>(nAccepted)/(static_cast<double>(N)*static_cast<double>(nRepeats));
524 }
525
526 //Normalise the weights
527 pPopulation.SetLogWeight(pPopulation.GetLogWeight() - CalcLogNC());
528
529 //Perform any final updates to the additional algorithm parameters.
530 pAdapt->updateEnd(algParams,pPopulation);
531
532 //Finally, the current particle set should be appended to the historical process.
533 if(htHistoryMode != HistoryType::NONE){
534 History.clear();
535 historyelement<Space> histel;
536 switch(htHistoryMode) {
537 case HistoryType::RAM:
538 histel.Set(N, pPopulation, nAccepted, nRepeats, historyflags(nResampled));
539 break;
540 case HistoryType::AL:
541 histel.Set(N, pPopulation, nAccepted, nRepeats, historyflags(nResampled), uRSIndices);
542 break;
544 case HistoryType::NONE:
545 break;
546 }
547 History.push_back(histel);
548 }
549 return;
550 }

559 template <class Space, class Params>
560 double sampler<Space,Params>::Integrate(double(*pIntegrand)(const Space&,void*), void * pAuxiliary)
561 {
562 long double rValue = 0;
563 long double wSum = expl(stableLogSumWeights(pPopulation.GetLogWeight()));
564 for(int i =0; i < N; i++)
565 {
566 rValue += expl(pPopulation.GetLogWeightN(i)) * pIntegrand(pPopulation.GetValueN(i), pAuxiliary);
567 }
568
569 rValue /= wSum;
570 return static_cast<double>(rValue);
571 }

602 template <class Space, class Params>
603 double sampler<Space,Params>::IntegratePathSampling(PathSamplingType::Enum PStype, double (*pIntegrand)(long,const Space &, void*), double (*pWidth)(long,void*), void* pAuxiliary)
604 {
605 if(htHistoryMode == HistoryType::NONE)
606 throw SMC_EXCEPTION(SMCX_MISSING_HISTORY, "The path sampling integral cannot be computed as the history of the system was not stored.");
607
608 // historyelement<Space> histel;
609 // histel.Set(N, pPopulation, nAccepted, nRepeats, historyflags(nResampled));
610 // History.push_back(histel);
611
612
613 long lTime = 1;
614 long double rValue = 0.0;
615 typename std::vector<historyelement<Space> >::const_iterator it;
616
617 switch(PStype) {
618 case PathSamplingType::RECTANGLE:
619 {
620 for(it = ++History.begin(); it!=History.end(); it++){
621 rValue += it->Integrate(lTime, pIntegrand, pAuxiliary) * static_cast<long double>(pWidth(lTime,pAuxiliary));
622 lTime++;
623 }
624 break;
625 }
626
627
628 case PathSamplingType::TRAPEZOID1:
629 {
630
631 long double previous_expt = History.begin()->Integrate(0,pIntegrand,pAuxiliary);
632 long double current_expt;
633 for(it = ++History.begin(); it!=History.end(); it++){
634 current_expt = it->Integrate(lTime, pIntegrand, pAuxiliary);
635 rValue += static_cast<long double>(pWidth(lTime,pAuxiliary))/2.0 * (previous_expt + current_expt) ;
636 lTime++;
637 previous_expt = current_expt;
638 }
639
640 break;
641 }
642
643 case PathSamplingType::TRAPEZOID2:
644 default:
645 {
646 long double previous_expt = History.begin()->Integrate(0,pIntegrand,pAuxiliary);
647 long double previous_var = History.begin()->Integrate_Var(0,pIntegrand,previous_expt,pAuxiliary);
648 long double current_expt;
649 long double current_var;
650 long double width = 0.0;
651 for(it = ++History.begin(); it!=History.end(); it++){
652 current_expt = it->Integrate(lTime, pIntegrand, pAuxiliary);
653 current_var = it->Integrate_Var(lTime, pIntegrand, current_expt, pAuxiliary);
654 width = static_cast<long double>(pWidth(lTime,pAuxiliary));
655 rValue += width/2.0 * (previous_expt + current_expt) - std::pow(width,2.0)/12.0*(current_var - previous_var);
656 lTime++;
657 previous_expt = current_expt;
658 previous_var = current_var;
659 }
660
661 break;
662 }
663
664 }
665
666 // History.pop_back();
667
668 return static_cast<double>(rValue);
669 }

779 template <class Space, class Params>
780 void sampler<Space,Params>::Resample(ResampleType::Enum lMode)
781 {
782 //Resampling is done in place.
783 int uMultinomialCount;
784
785 //First obtain a count of the number of children each particle has.
786 switch(lMode) {
787 case ResampleType::MULTINOMIAL:
788 //Sample from a suitable multinomial vector
789 dRSWeights = exp(pPopulation.GetLogWeight() - stableLogSumWeights(pPopulation.GetLogWeight()));
790 rmultinom(static_cast<int>(N), dRSWeights.memptr(), static_cast<int>(N), uRSCount.memptr());
791 break;
792
793 case ResampleType::RESIDUAL:
794 dRSWeights = exp(log(static_cast<double>(N)) + pPopulation.GetLogWeight() - stableLogSumWeights(pPopulation.GetLogWeight()));
795 uRSIndices = arma::zeros<arma::Col<unsigned int> >(static_cast<int>(N));
796 for(int i = 0; i < N; ++i)
797 uRSIndices(i) = static_cast<unsigned int>(floor(dRSWeights(i)));
798 dRSWeights = dRSWeights - uRSIndices;
799 dRSWeights = dRSWeights/sum(dRSWeights);
800 uMultinomialCount = N - arma::sum(uRSIndices);
801 rmultinom(uMultinomialCount, dRSWeights.memptr(), static_cast<int>(N), uRSCount.memptr());
802 uRSCount += arma::conv_to<arma::Col<int> >::from(uRSIndices);
803 break;
804
805 case ResampleType::STRATIFIED:
806 default:
807 {
808 // Procedure for stratified sampling
809 //Generate a random number between 0 and 1/N
810 double dRand = R::runif(0,1.0 / static_cast<double>(N));
811 arma::vec dWeightCumulative = arma::cumsum(exp(pPopulation.GetLogWeight() - stableLogSumWeights(pPopulation.GetLogWeight())));
812 int j = 0, k = 0;
813 uRSCount = arma::zeros<arma::Col<int> >(static_cast<int>(N));
814 //while(j < N) {
815 while(k < N) {
816 while((dWeightCumulative(k) - dRand) > static_cast<double>(j)/static_cast<double>(N) && j < N) {
817 uRSCount(k)++;
818 j++;
819 dRand = R::runif(0,1.0 / static_cast<double>(N));
820 }
821 k++;
822 }
823 break;
824 }
825
826 case ResampleType::SYSTEMATIC:
827 {
828 // Procedure for stratified sampling but with a common RV for each stratum
829 //Generate a random number between 0 and 1/N
830 double dRand = R::runif(0,1.0 / static_cast<double>(N));
831 int j = 0, k = 0;
832 uRSCount = arma::zeros<arma::Col<int> >(static_cast<int>(N));
833 arma::vec dWeightCumulative = arma::cumsum(exp(pPopulation.GetLogWeight() - stableLogSumWeights(pPopulation.GetLogWeight())));
834 //while(j < N) {
835 while(k < N) {
836 while((dWeightCumulative(k) - dRand) > static_cast<double>(j)/static_cast<double>(N) && j < N) {
837 uRSCount(k)++;
838 j++;
839 }
840 k++;
841 }
842 break;
843 }
844 }
845
846 uRSIndices = arma::zeros<arma::Col<unsigned int> >(static_cast<int>(N));
847 //Map count to indices to allow in-place resampling
848 for (int i=0, j=0; i<N; ++i) {
849 if (uRSCount(i)>0) {
850 uRSIndices(i) = i;
851 while (uRSCount(i)>1) {
852 while (uRSCount(j)>0) ++j; // find next free spot
853 uRSIndices(j++) = i; // assign index
854 --uRSCount(i); // decrement number of remaining offsprings
855 }
856 }
857 }
858
859 //Perform the replication of the chosen.
860 for(int i = 0; i < N ; ++i) {
861 if(uRSIndices(i) != static_cast<unsigned int>(i)){
862 pPopulation.SetValueN( pPopulation.GetValueN(static_cast<int>(uRSIndices(i))) ,i);
863 }
864 }
865
866 //Set equal normalised weights
867 pPopulation.SetLogWeight(- log(static_cast<double>(N))*arma::ones(N));
868 }
