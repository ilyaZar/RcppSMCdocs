158 template <class Space>
159 long double historyelement<Space>::Integrate(long lTime, double (*pIntegrand)(long,const Space&,void*), void* pAuxiliary) const
160 {
161 long double rValue = 0;
162 long double wSum = expl(stableLogSumWeights(pop.GetLogWeight()));
163 for(long i =0; i < number; i++)
164 {
165 rValue += expl(pop.GetLogWeightN(i)) * static_cast<long double>(pIntegrand(lTime, pop.GetValueN(i), pAuxiliary));
166 }
167
168 rValue /= wSum;
169 return rValue;
170 }

176 template <class Space>
177 long double historyelement<Space>::Integrate_Var(long lTime, double (*pIntegrand)(long,const Space &,void*), double Expectation, void* pAuxiliary) const
178 {
179 long double rValue = 0;
180 long double wSum = expl(stableLogSumWeights(pop.GetLogWeight()));
181 for(long i =0; i < number; i++)
182 {
183 rValue += expl(pop.GetLogWeightN(i)) * std::pow(static_cast<long double>(pIntegrand(lTime, pop.GetValueN(i), pAuxiliary) - Expectation),2.0);
184 }
185
186 rValue /= wSum;
187 return rValue;
188 }
