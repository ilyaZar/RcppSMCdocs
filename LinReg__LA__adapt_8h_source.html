68 //A derived class for adaptation which adapts parameters of type smc::staticModelAdapt
69 class rad_adapt:
70 public smc::adaptMethods<rad_state,smc::staticModelAdapt>
71 {
72 public:
73
74 void updateForMove(smc::staticModelAdapt & param, const smc::population<rad_state> & pop) {
75 unsigned long N = pop.GetNumber();
76 arma::vec loglike(N);
77 for (unsigned int i=0; i<N; i++){
78 loglike(i) = pop.GetValueN(i).loglike;
79 }
80 param.ChooseTemp(pop.GetLogWeight(),loglike,rho*N);
81 }
82
83 void updateForMCMC(smc::staticModelAdapt & param, const smc::population<rad_state> & pop, double acceptProb, int nResampled, int & nRepeats) {
84 if (nResampled == 0) {
85 nRepeats = 0;
86 } else {
87 nRepeats = param.calcMcmcRepeats(acceptProb, 0.99, 10, 1000);
88 }
89
90 arma::mat thetaMat(pop.GetNumber(),3);
91 for (long i=0; i<pop.GetNumber(); i++){
92 thetaMat.row(i) = pop.GetValueN(i).theta.t();
93 }
94 param.calcCholCov(thetaMat,pop.GetLogWeight());
95 }
96
97 ~rad_adapt() {};
98
99 };
