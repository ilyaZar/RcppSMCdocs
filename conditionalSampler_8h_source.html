161 template <class Space, class Params>
162 void conditionalSampler<Space,Params>::Initialise(void)
163 {
164 T = 0;
165 dlogNCPath = 0.0;
166
167 //Set the initial values and log weights of the particles
168 std::vector<Space> InitVal(N);
169 arma::vec InitWeights(N);
170 pPopulation = population<Space>(InitVal,InitWeights);
171 pMoves->DoInit(pPopulation,N,algParams);
172
173 //Initialise the conditonal trajectory:
174 //1. Sample uniformly initial period, T = 0, conditional index
175 // referenceTrajectoryIndices.at(T) = floor(unif_rand()*static_cast<double>(N));
176 referenceTrajectoryIndices.at(T) = 0;
177 //2. Set first particle coordinate to conditional value at above index,
178 //and re-weight using the DoConditionalMove-function (that, despite its name, works at initialization, T=0, as well as moves for subsequent T>=1 iterations)
179 pMoves->DoConditionalMove(T,pPopulation,referenceTrajectory[T],referenceTrajectoryIndices.at(T),algParams);
180
181 //Scaling weights by 1/N (for evidence computation)
182 pPopulation.SetLogWeight(pPopulation.GetLogWeight() - log(static_cast<double>(N)));
183
184 //Estimate the normalising constant
185 dlogNCIt = CalcLogNC();
186 dlogNCPath += dlogNCIt;
187
188 //Normalise the weights
189 pPopulation.SetLogWeight(pPopulation.GetLogWeight() - dlogNCIt);
190
191 //Check if the ESS is below some reasonable threshold and resample if necessary.
192 //A mechanism for setting this threshold is required.
193 double ESS = GetESS();
194 if(ESS < dResampleThreshold) {
195 nResampled = 1;
196 conditionalResample(rtResampleMode);
197 }
198 else {
199 nResampled = 0;
200 if(htHistoryMode == HistoryType::AL) {
201 uRSIndices = arma::linspace<arma::Col<unsigned int>>(0, N - 1, N);
202 //No resampling: set conditional index to previous iteration.
203 referenceTrajectoryIndices.at(T + 1) = referenceTrajectoryIndices.at(T);
204 }
205 }
206
207 //Normalise the weights
208 pPopulation.SetLogWeight(pPopulation.GetLogWeight() - CalcLogNC());
209
210 //Finally, the current particle set should be appended to the historical process.
211 if(htHistoryMode != HistoryType::NONE){
212 History.clear();
213 historyelement<Space> histel;
214 switch(htHistoryMode) {
215 case HistoryType::RAM:
216 histel.Set(N, pPopulation, historyflags(nResampled));
217 break;
218 case HistoryType::AL:
219 histel.Set(N, pPopulation, historyflags(nResampled), uRSIndices);
220 break;
222 case HistoryType::NONE:
223 break;
224 }
225 History.push_back(histel);
226 }
227 return;
228 }

305 {

311 MoveReferenceParticle();

315 dlogNCPath += dlogNCIt;

318 pPopulation.SetLogWeight(pPopulation.GetLogWeight() - dlogNCIt);

332 referenceTrajectoryIndices.at(T + 1) = referenceTrajectoryIndices.at(T);
333 }
334 }
335
336 //Normalise the weights
337 pPopulation.SetLogWeight(pPopulation.GetLogWeight() - CalcLogNC());
338

375 template <class Space, class Params>
376 void conditionalSampler<Space, Params>::conditionalResample(ResampleType::Enum lMode)
377 {
378 //Conditional resampling performed following the algorithms outlined in Appendix C of the paper "...".
379
380 //Initialize container for ancestor indices.
381 uRSIndices = arma::zeros<arma::Col<unsigned int> >(static_cast<int>(N));
382
383 switch(lMode) {
384 case ResampleType::MULTINOMIAL:
385 default:
386 {
387 //Algorithm 3
388 //Step 0:
389 //Sample conditional index K_t from appropriate version of the "lambda" distribution i.e. uniformly on {1,...,N} in case of Multinmial resampling.
390 // long Kt = floor(unif_rand()*static_cast<double>(N)); // sample lamba(k_{t}|w_{t-1}, k_{t-1})=1/N
391 //Set conditional index K_t to zero implementing a sepcial case that works for conditional multinomial resampling (saving one draw compared to the implementation above).
392 long Kt = 0;
393 referenceTrajectoryIndices.at(T + 1) = Kt; // update referenceTrajectoryIndices with newly sampled K_t index.
394 //Step 1:
395 //Connect the "chosen" ancestor index to previous reference trajectory: A_{t - 1}^{K_t} = K_{t - 1}
396 uRSIndices.at(Kt) = referenceTrajectoryIndices.at(T);
397 //Step 2:
398 //Sample remaining ancestors A_{t - 1}^{-K_t} i.i.d. from a categorical distribution.
399 // 2.1. Generate weights for categorical distribution.
400 dRSWeights = exp(pPopulation.GetLogWeight() - stableLogSumWeights(pPopulation.GetLogWeight()));
401 // 2.2 Sample remaining N-1 ancestor indices from {1,...,N}.
402 Rcpp::IntegerVector tmpAncestorIndices(N - 1);
403 tmpAncestorIndices = Rcpp::sample(N, N - 1, true, Rcpp::as<Rcpp::NumericVector>(Rcpp::wrap(dRSWeights))) - 1;
404 // 2.3. Assign ancestor indices to offspring {1,...,N}\{Kt}.
405 std::vector<unsigned int> tmpIterator(N);
406 std::iota(tmpIterator.begin(), tmpIterator.end(), 0); // define appropriate tmpIterator as a sequence from 0 to N-1
407 tmpIterator.erase(tmpIterator.begin() + Kt); //exclude the previoiusly sampled conditional index K_t.
408 long intIncrement = 0;
409 // assign sampled ancestors to children
410 for (int i : tmpIterator) {
411 uRSIndices.at(i) = tmpAncestorIndices(intIncrement);
412 ++intIncrement;
413 }
414 break;
415 }
416 case ResampleType::STRATIFIED:
417 {
418 //Algorithm 4
419 //Step 0:
420 //Sample conditional index K_t from appropriate version of the "lambda" distribution i.e. the distribution over the stratum with \lambda(k_t|w_{t - 1}^{1:N}, k_{t - 1})=p_{t - 1}^{k_{t - 1}}(k_t)/W_{t - 1}^{k_{t - 1}} in case of stratified resampling.
421 //0.0 Housekeeping for step 0.1 - 0.5
422 // k_{t - 1}
423 int kt1 = referenceTrajectoryIndices.at(T);
424 // sum_{k=1}^{k_{t-1}}\hat{w}_{t-1}^k
425 double minWeights = 0;
426 // sum_{k=1}^{k_{t-1} - 1}\hat{w}_{t-1}^k
427 double maxWeights = 0;
428 // For N different strata (N being the number of particles)
429 // all strata boundaries: 0/N,...,c/N,...,N/N for c = 0,...,N
430 arma::Col<double> stBoundsAll = arma::linspace(0.0, 1.0, N + 1);
432 arma::Col<double> stBoundsUpper = stBoundsAll.tail(N);
434 arma::Col<double> stBoundsLower = stBoundsAll.head(N);
436 arma::Col<double> strataWeights(N);
437 strataWeights.fill(0.0);
439 Rcpp::NumericVector lambdaWeightsStratified(N);
440 //0.1. Calculate normalized particle weights and cumulative normalized weights.
441 dRSWeights = exp(pPopulation.GetLogWeight() - stableLogSumWeights(pPopulation.GetLogWeight()));
442 arma::vec dRSWeightsCumulative = arma::cumsum(dRSWeights);
443 // 0.2. Calculate min() and max() parts of strata weights
444 minWeights = dRSWeightsCumulative.at(kt1);
445 if(kt1 > 0){
446 maxWeights = dRSWeightsCumulative.at(kt1 - 1);
447 } else {
448 maxWeights = 0;
449 }
450 // 0.3. Calculate weight of particle k_{t-1} in all strata
451 // k_t=1,...,N i.e. p_{t-1}^{k_{t-1}}(k_t)
452 for(int i = 0; i < N; ++i) {
453 strataWeights.at(i) = std::max(std::min(minWeights, stBoundsUpper.at(i)) - std::max(maxWeights, stBoundsLower.at(i)), 0.0);
454 }
455 // 0.4. Calculate \lambda(k_t|.) distribution.
456 lambdaWeightsStratified = Rcpp::wrap(strataWeights/dRSWeights.at(kt1));
457 // 0.5. Sample K_t from 0.4.
458 long Kt = Rcpp::sample(N, 1, false, lambdaWeightsStratified)[0] - 1;
459 //Step 1:
460 //Connect the "chosen" ancestor index to previous reference trajectory: A_{t - 1}^{K_t} = K_{t - 1}
461 uRSIndices.at(Kt) = kt1;
462 //Step 2:
463 //Calculation of empirical distribution function F_{t - 1}^N(i) is done and equal to computation of cumulative normalized weights stored in dRSWeightsCumulative.
464 //Step 3: Generate ancestor indices and assign to offspring indices {1,...,N}\{Kt}.
465 std::vector<unsigned int> tmpIterator(N);
466 std::iota(tmpIterator.begin(), tmpIterator.end(), 0); // define appropriate tmpIterator as a sequence from 0 to N-1
467 tmpIterator.erase(tmpIterator.begin() + Kt); // exclude the previoiusly sampled conditional index K_t
468 long minimalJ = 0;
469 // generate ancestor index and assign to offspring
470 double tmpUnifRnd;
471 for (int i : tmpIterator) {
472 tmpUnifRnd = unif_rand();
473 tmpUnifRnd += i;//mathematical 'i' starts at 1; here at 0
474 tmpUnifRnd /= N;
475
476 minimalJ += arma::conv_to<long>::from(arma::find(dRSWeightsCumulative.tail(N - minimalJ) > tmpUnifRnd, 1, "first"));
477 uRSIndices.at(i) = minimalJ;
478 }
479 break;
480 }
481 case ResampleType::SYSTEMATIC:
482 {
483 //Algorithm 5
484 //Step 0:
485 //Sample conditional index K_t from appropriate version of the "lambda" distribution i.e. the distribution over the stratum with \lambda(k_t|w_{t - 1}^{1:N}, k_{t - 1})=p_{t - 1}^{k_{t - 1}(k_t)/W_{t - 1}^{k_{t - 1}} (same as stratified resampling).
486 //0.0 Housekeeping for step 0.1 - 0.5
487 // k_{t - 1}
488 int kt1 = referenceTrajectoryIndices.at(T);
489 // sum_{k=1}^{k_{t-1}}\hat{w}_{t-1}^k
490 double minWeights = 0;
491 // sum_{k=1}^{k_{t-1} - 1}\hat{w}_{t-1}^k
492 double maxWeights = 0;
493 // For N different strata (N being the number of particles)
494 // all strata boundaries: 0/N,...,c/N,...,N/N for c = 0,...,N
495 arma::Col<double> stBoundsAll = arma::linspace(0.0, 1.0, N + 1);
497 arma::Col<double> stBoundsUpper = stBoundsAll.tail(N);
499 arma::Col<double> stBoundsLower = stBoundsAll.head(N);
501 arma::Col<double> strataWeights(N);
502 strataWeights.fill(0.0);
504 Rcpp::NumericVector lambdaWeightsStratified(N);
505 // 0.1. Calculate normalized particle weights and cumulative normalized weights.
506 dRSWeights = exp(pPopulation.GetLogWeight() - stableLogSumWeights(pPopulation.GetLogWeight()));
507 arma::vec dRSWeightsCumulative = arma::cumsum(dRSWeights);
508 // 0.2. Calculate min() and max() parts of strata weights
509 minWeights = dRSWeightsCumulative.at(kt1);
510 if(kt1 > 0){
511 maxWeights = dRSWeightsCumulative.at(kt1 - 1);
512 } else {
513 maxWeights = 0;
514 }
515 // 0.3. Calculate weight of particle k_{t-1} in all strata
516 // k_t=1,...,N i.e. p_{t-1}^{k_{t-1}}(k_t)
517 for(int i = 0; i < N; ++i) {
518 strataWeights.at(i) = std::max(std::min(minWeights, stBoundsUpper.at(i)) - std::max(maxWeights, stBoundsLower.at(i)), 0.0);
519 }
520 // 0.4. Calculate \lambda(k_t|.) distribution.
521 lambdaWeightsStratified = Rcpp::wrap(strataWeights/dRSWeights.at(kt1));
522 // 0.5. Sample K_t from 0.4.
523 long Kt = Rcpp::sample(N, 1, false, lambdaWeightsStratified)[0] - 1;
524 //Step 1:
525 //Connect the "chosen" ancestor index to previous reference trajectory: A_{t - 1}^{K_t} = K_{t - 1}
526 uRSIndices.at(Kt) = referenceTrajectoryIndices.at(T);
527 //Step 2:
528 //Calculation of empirical distribution function F_{t - 1}^N(i) is done and equal to computation of cumulative normalized weights stored in dRSWeightsCumulative.
529 //Step 3: Generate ancestor indices and sssign to offspring indices {1,...,N}\{Kt}.
530 std::vector<unsigned int> tmpIterator(N);
531 std::iota(tmpIterator.begin(), tmpIterator.end(), 0); // define appropriate tmpIterator as a sequence from 0 to N-1
532 tmpIterator.erase(tmpIterator.begin() + Kt); // exclude the previoiusly sampled conditional index K_t
533 //precompute necessary uniform random variable before assignment
534 double tmpVUpperBound = dRSWeightsCumulative.at(Kt);
535 double tmpVLowerBound;
536 if(Kt == 0) {
537 tmpVLowerBound = 0;
538 } else {
539 tmpVLowerBound = dRSWeightsCumulative.at(Kt - 1);
540 }
541 double tmpV = R::runif(tmpVLowerBound, tmpVUpperBound);
542 tmpV = N * tmpV - std::floor(N * tmpV);
543 double tmpU;
544 long minimalJ = 0;
545 // generate ancestor index and assign to offspring
546 for (int i : tmpIterator) {
547 tmpU = tmpV + i;//mathematical 'i' at 1; here it starts at 0
548 tmpU /= N;
549
550 minimalJ += arma::conv_to<long>::from(arma::find(dRSWeightsCumulative.tail(N - minimalJ) > tmpU, 1, "first"));
551 uRSIndices.at(i) = minimalJ;
552 }
553 break;
554 }
555 case ResampleType::RESIDUAL:
556 {
557 //Algorithm 6
558 //Step 0:
559 //Initialize container for implementation; pre-compute normalized/unnormalized residual weights; perform deterministic component assignment and compute other quantities
560 // 0.1. Container setup
561 int l = 0; //Counts the number of deterministically assigned offspring.
562 double kt1 = referenceTrajectoryIndices.at(T); // k_{t-1}
563 int expectedNumberOffspring = 0; //Integer part of N*W_{t - 1}.
564 arma::Col<double> dRSWeightsResidual(N);
565 arma::Col<double> dRSWeightsResidualNormalized(N);
566 arma::Col<unsigned int> DsetCurrent; //D_i
567 int CardDsetCurrent = 0;//|D_i|
568 arma::Col<unsigned int> DsetKtMinus1; //D_{K_{t - 1}}
569 arma::Col<unsigned int> DsetDeterministic; //The set of indices that are sampled deterministically.
570 arma::Col<unsigned int> DsetStochastic; //The set of indices that are sampled stochastically.
571 arma::Col<unsigned int> tmpIterator = arma::linspace<arma::Col<unsigned int> >(0, N - 1, N);
572 // 0.2. Calculate normalized particle weights.
573 dRSWeights = exp(pPopulation.GetLogWeight() - stableLogSumWeights(pPopulation.GetLogWeight()));
574
575 for (int i : tmpIterator) {
576 //Compute (integer part of) expected number of offspring
577 expectedNumberOffspring = std::floor(N * dRSWeights.at(i));
578 //Generate D_i set:
579 if(expectedNumberOffspring > 0) {
580 //Set D_i={l + 1, ..., l + expectedNumberOffspring}
581 DsetCurrent = arma::linspace<arma::Col<unsigned int> >(l, l + expectedNumberOffspring - 1, expectedNumberOffspring);
582 // Set Card(D_i)=length(D_i) = floor(N*W_{t-1});
583 CardDsetCurrent = expectedNumberOffspring;
584 // Set l += Card(D_i);
585 l += CardDsetCurrent;
586 } else {
587 //Set D_i=EMPTYSET
588 //Set Card(D_i)= 0;
589 DsetCurrent.reset();
590 CardDsetCurrent = 0;
591 }
592 //Assign deterministic offpring indices for each i={1,...,N}
593 for (int j : DsetCurrent) {
594 uRSIndices.at(j) = i;
595 }
596 //Compute unnormalized residual weights \hat{W}_{t-1}^i.
597 dRSWeightsResidual.at(i) = dRSWeights.at(i) * N - static_cast<double>(CardDsetCurrent)/N;
598 //Check if convenient way of handling the conditioning path
599 //is possible: if(i == K_{t-1}) store D_{K_{t-1}}
600 //and check K_t \notin D_{K_{t-1}} later!
601 if(i == static_cast<int>(referenceTrajectoryIndices.at(T))) {
602 DsetKtMinus1 = DsetCurrent;
603 }
604 }
605 //Define deterministic and stochastic sets for index assignment
606 DsetDeterministic = arma::linspace<arma::Col<unsigned int> >(0, l - 1, l);
607 if (l < N) {
608 DsetStochastic = arma::linspace<arma::Col<unsigned int> >(l, N - 1, N - l);
609 } else {
610 DsetStochastic = arma::linspace<arma::Col<unsigned int> >(N - 1, N - 1, 1);
611 }
612 //Compute normalized residual weights \tilde{W}_{t-1}^i.
613 dRSWeightsResidualNormalized = dRSWeightsResidual/arma::sum(dRSWeightsResidual);
614 //Step 1:
615 //Sample K_t from appropriate lambda distribution
616 // 1.1 Initializes container
617 arma::Col<double> lambdaProbs(N);
618 lambdaProbs.zeros();
619 double prob1 = 1.0/(N * dRSWeights.at(kt1));
620 double prob2 = dRSWeightsResidual.at(kt1)/(N * dRSWeights.at(kt1) * (N - l));
621 //Defines the complement set of DsetKtMinus1 relative to {1,...,N} i.e. {1,...,N}\DsetKtMinus1
622 // 1.2 Fills a vector with probabilities to sample from \lambda(k_t|k_{t - 1}, x_{t - 1}^{1:N})
623 lambdaProbs.elem(DsetKtMinus1).fill(prob1);
624 lambdaProbs.elem(DsetStochastic).fill(prob2);
625 // 1.3 Sample Kt from eq. 4 on p.8 of the WP
626 long Kt = Rcpp::sample(N, 1, false, Rcpp::wrap(lambdaProbs))[0] - 1;
627 referenceTrajectoryIndices(T + 1) = Kt;
628 // 1.4 If Kt \notin D_{K_{t-1}}
629 if (DsetKtMinus1.size() == 0) { //the set is empty
630 uRSIndices.at(Kt) = kt1;
631 } else if (Kt < arma::as_scalar(DsetKtMinus1.head(1)) && Kt > arma::as_scalar(DsetKtMinus1.tail(1))) {//Kt notin (ordered) set
632 uRSIndices.at(Kt) = kt1;
633 }
634 //Step 2:
635 //Compute residual ancestor indices via sampling from categorical distribution
636 if (l < N) {
637 for(int k : DsetStochastic) {
638 if (k == Kt) continue; // skip since A^{K_t} already set
639 uRSIndices.at(k) = Rcpp::sample(N, 1, false, Rcpp::wrap(dRSWeightsResidualNormalized))[0] - 1;
640 }
641 }
642 break;
643 }
644 }
645 //Copy values of pre-resampling particle set.
646 std::vector<Space> populationValueCopy(pPopulation.GetValue().begin(), pPopulation.GetValue().end());
647 //Perform the replication of the chosen.
648 for(int i = 0; i < N ; ++i) {
649 if(uRSIndices(i) != static_cast<unsigned int>(i)){
650 pPopulation.SetValueN(populationValueCopy[static_cast<int>(uRSIndices(i))], i);
651 }
652 }
653 //After conditional resampling is implemented: a final step is to set equal normalised weights.
654 pPopulation.SetLogWeight(- log(static_cast<double>(N))*arma::ones(N));
655 }
