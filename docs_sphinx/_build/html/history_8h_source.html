159  long double historyelement<Space>::Integrate(long lTime, double (*pIntegrand)(long,const Space&,void*), void* pAuxiliary) const
160  {
161  long double rValue = 0;
162  long double wSum = expl(stableLogSumWeights(pop.GetLogWeight()));
163  for(long i =0; i < number; i++)
164  {
165  rValue += expl(pop.GetLogWeightN(i)) * static_cast<long double>(pIntegrand(lTime, pop.GetValueN(i), pAuxiliary));
166  }
167 
168  rValue /= wSum;
169  return rValue;
170  }
