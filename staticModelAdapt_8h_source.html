3// staticModelAdapt.h: A class containing parameters and functions to update
4// these parameters in the context of static Bayesian models. The methods to
5// estimate the empirical covariance and its Cholesky decomposition are applicable
6// for applications where the particle values are a vector of doubles. The methods
7// to compute the next 'temperature' are relevant to likelihood annealing SMC where the
8// power posteriors are defined by P_t(theta|y) \propto P(y|theta)^\gamma_t P(theta).
9// Here y is observed data, theta denotes the parameters of the model and gamma
10// denotes the 'temperatures' which start at 0 (the prior) and finish at 1
11// (the posterior).

81 inline double staticModelAdapt::CESSdiff(const arma::vec & logweight, const arma::vec & loglike, double tempDiff, double desiredCESS){
82 double logsum1 = stableLogSumWeights(logweight + tempDiff*loglike);
83 double logsum2 = stableLogSumWeights(logweight + 2*tempDiff*loglike);
84
85 return exp(log(static_cast<double>(logweight.n_rows)) + 2*logsum1 - logsum2) - desiredCESS;
86 }

89 inline double staticModelAdapt::bisection(double curr, const arma::vec & logweight, const arma::vec & loglike, double desiredCESS, double epsilon){
90 double a = curr;
91 double b = 1.0;
92 double f_a = CESSdiff(logweight,loglike,a-curr,desiredCESS);
93 double f_b = CESSdiff(logweight,loglike,b-curr,desiredCESS);
94 if (f_a*f_b>0){
95 Rcpp::stop("Bisection method to choose the next temperature failed");
96 } else{
97 double m, f_m, err;
98 m = (a+b)/2.0;
99 f_m = CESSdiff(logweight,loglike,m-curr,desiredCESS);
100 err = 10.0;
101 while (err > epsilon){
102 if (f_m<0.0){
103 b = m;
104 f_b = f_m;
105 } else{
106 a = m;
107 f_a = f_m;
108 }
109 m = (a+b)/2.0;
110 f_m = CESSdiff(logweight,loglike,m-curr,desiredCESS);
111 err = std::abs(f_m);
112 }
113 return m;
114 }
115 }

123 inline void staticModelAdapt::ChooseTemp(const arma::vec & logweight, const arma::vec & loglike, double desiredCESS, double epsilon) {
124 double temp_curr = temp.back();
125 if (CESSdiff(logweight,loglike,1.0-temp_curr,desiredCESS)>=-epsilon){
126 temp.push_back(1.0);
127 } else {
128 temp.push_back(bisection(temp_curr, logweight, loglike, desiredCESS, epsilon));
129 }
130 }

137 inline void staticModelAdapt::calcEmpCov(const arma::mat & theta, const arma::vec & logweight){
138 int N = logweight.n_rows;
139 arma::vec normWeights = exp(logweight - stableLogSumWeights(logweight));
140
141 arma::mat diff = theta - arma::ones(N,1)*arma::mean(theta,0);
142 empCov = diff.t()*diagmat(normWeights)*diff;
143 }

150 inline void staticModelAdapt::calcCholCov(const arma::mat & theta, const arma::vec logweight){
151 calcEmpCov(theta,logweight);
152 cholCov = arma::chol(empCov);
153 }

161 inline int staticModelAdapt::calcMcmcRepeats(double acceptProb, double desiredAcceptProb, int initialN, int maxReps){
162 if (acceptProb + 1.0 <= 1e-9){
163 return initialN;
164 } else if (acceptProb - 1.0 >= -1e-9){
165 return 1;
166 } else if (acceptProb <= 1e-9){
167 return maxReps;
168 } else {
169 return std::min(maxReps,static_cast<int>(std::ceil(log(1.0-desiredAcceptProb)/log(1.0-acceptProb))));
170 }
171 }
